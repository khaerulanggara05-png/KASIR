    items.forEach((el,i)=>el.classList.toggle('search-focused', i===searchFocusIdx));

    items[searchFocusIdx].scrollIntoView({block:'nearest'});

  } else if(e.key === 'ArrowUp'){

    e.preventDefault();

    searchFocusIdx = Math.max(searchFocusIdx-1, 0);

    items.forEach((el,i)=>el.classList.toggle('search-focused', i===searchFocusIdx));

    items[searchFocusIdx].scrollIntoView({block:'nearest'});

  } else if(e.key === 'Enter'){

    e.preventDefault();

    if(searchFocusIdx >= 0 && items[searchFocusIdx]){

      items[searchFocusIdx].click();

    } else if(items.length === 1){

      items[0].click();

    }

  } else if(e.key === 'Escape'){

    document.getElementById('searchQ').value = '';

    filterProd();

  }

}

// Hitung subtotal item setelah diskon per item

function recalcItem(item){

  const base = item.harga * item.qty;

  if(!item.disType || item.disVal===0){

    item.subtotal = base;

    item.disAmt   = 0;

  } else if(item.disType==='pct'){

    item.disAmt   = Math.round(base * Math.min(item.disVal,100) / 100);

    item.subtotal = base - item.disAmt;

  } else {

    item.disAmt   = Math.min(item.disVal, base);

    item.subtotal = base - item.disAmt;

  }

}

function calcTotal(){

  const sub    = order.reduce((a,b)=>a+b.harga*b.qty,0);

  const totalDis = order.reduce((a,b)=>a+(b.disAmt||0),0);

  document.getElementById('pDiskonVal').textContent = totalDis>0 ? `− ${idr(totalDis)}` : `− Rp 0`;

  document.getElementById('pTotal').textContent     = idr(total);

  calcChange();

}

function calcChange(){

  const total = Number(document.getElementById('pTotal').textContent.replace(/\D/g,''));

  const bayar = parseBayar();

  const row   = document.getElementById('changeRow');

  if(bayar===0){

    document.getElementById('pChange').textContent='Rp 0';

    row.classList.remove('kurang');

    return;

  }

  const kmbl = bayar - total;

  if(kmbl >= 0){

    document.getElementById('pChange').textContent = idr(kmbl);

    row.classList.remove('kurang');

  } else {

    document.getElementById('pChange').textContent = '− '+idr(Math.abs(kmbl));

    row.classList.add('kurang');

  }

}

// Parse harga dari berbagai format:

// 450000 → 450000 | 450.000 → 450000 | 450,000 → 450000

// Rp 450.000 → 450000 | 450.5 (desimal USD) → 450

function parseHarga(raw){

  if(!raw && raw !== 0) return 0;

  // Jika sudah number langsung (dari Excel cell numerik)

  if(typeof raw === 'number') return Math.round(raw);

  let s = raw.toString().trim();

  // Buang simbol mata uang dan spasi

  s = s.replace(/[Rrp$€\s]/gi,'');

  // Deteksi format: jika ada titik DAN koma, titik = pemisah ribuan

  // Jika hanya titik: cek apakah 3 digit di belakang (pemisah ribuan Indonesia)

  if(s.includes(',') && s.includes('.')){

    // Contoh: 1.450.000 atau 1,450,000

    // Buang semua pemisah ribuan, sisakan bagian desimal jika ada

    const lastComma = s.lastIndexOf(',');

    const lastDot   = s.lastIndexOf('.');

    if(lastDot > lastComma){

      // Format US: 1,450.00 — titik = desimal

      s = s.replace(/,/g,'');

    } else {

      // Format ID: 1.450,00 — koma = desimal, titik = ribuan

      s = s.replace(/\./g,'').replace(',','.');

    }

  } else if(s.includes('.')){

    // Hanya titik: 450.000 — cek apakah persis 3 digit setelah titik terakhir

    const parts = s.split('.');

    const last  = parts[parts.length-1];

    if(last.length === 3){

      // Pemisah ribuan Indonesia: 450.000 → 450000

      s = s.replace(/\./g,'');

    }

    // Kalau 450.50 (2 digit) → desimal, biarkan

  } else if(s.includes(',')){

    // Hanya koma: 450,000

    const parts = s.split(',');

    const last  = parts[parts.length-1];

    if(last.length === 3){

      s = s.replace(/,/g,'');

    } else {

      s = s.replace(',','.');

    }

  }

  const n = parseFloat(s);

  return isNaN(n) ? 0 : Math.round(n);

}

function handleCSV(file){

  if(!file)return;

  const r=new FileReader();

  r.onload=e=>{

    const lines=e.target.result.split(/\r?\n/).filter(l=>l.trim());

    const rows=[];

    for(let i=1;i<lines.length;i++){const c=lines[i].split(',');if(c.length<2)continue;const o=toRow(c);if(o.kode&&o.nama)rows.push(o);}

    sendRows(rows);

  };

  r.readAsText(file);

}

function handleXLSX(file){

  if(!file)return;

  if(typeof XLSX==='undefined'){toast('Library Excel belum siap','err');return;}

  const r=new FileReader();

  r.onload=e=>{

    try{

      const wb=XLSX.read(new Uint8Array(e.target.result),{type:'array'});

      const ws=wb.Sheets[wb.SheetNames[0]];

      const data=XLSX.utils.sheet_to_json(ws,{header:1,defval:''});

      if(data.length<2){toast('Sheet kosong','err');return;}

      const start=data[0]&&data[0][0]&&data[0][0].toString().toLowerCase().includes('kode')?1:0;

      const rows=[];

      for(let i=start;i<data.length;i++){const o=toRow(data[i]);if(o.kode&&o.nama)rows.push(o);}

      sendRows(rows);

    }catch(err){toast('Gagal baca Excel: '+err.message,'err');}

  };

  r.readAsArrayBuffer(file);

}

async function handlePDF(file){

  if(!file)return;

  if(typeof pdfjsLib==='undefined'){toast('Library PDF belum siap','err');return;}

  toast('Membaca PDF…','info');

  try{

    const ab=await file.arrayBuffer();

    const pdf=await pdfjsLib.getDocument({data:new Uint8Array(ab)}).promise;

    let txt='';

    for(let p=1;p<=pdf.numPages;p++){

      const page=await pdf.getPage(p);

      const tc=await page.getTextContent();

      const byY={};

      tc.items.forEach(item=>{const y=Math.round(item.transform[5]);if(!byY[y])byY[y]=[];byY[y].push(item.str.trim());});

      Object.keys(byY).map(Number).sort((a,b)=>b-a).forEach(y=>{txt+=byY[y].filter(s=>s).join('\t')+'\n';});

    }

    parsePdfTxt(txt);

  }catch(err){toast('Gagal baca PDF: '+err.message,'err');}

}

function parsePdfTxt(txt){

  const lines=txt.split('\n').filter(l=>l.trim());

  const rows=[];

  lines.forEach(line=>{

    const cols=line.split('\t').map(s=>s.trim()).filter(Boolean);

    if(cols.length<2)return;

    const h=cols[0].toString().toLowerCase();

    if(h.includes('kode')||h.includes('nama')||h.includes('code'))return;

    const o=toRow(cols);

    if(o.kode&&o.nama&&o.kode.length<=20)rows.push(o);

  });

  if(!rows.length){toast('Tidak bisa deteksi data dari PDF ini','err');return;}

  showPdfPreview(rows);

}

function applyHistFilter(){

  // Kalau ada perubahan filter manual, hapus quick button active

  const from = document.getElementById('filterDateFrom').value;

  histFiltered = histAllData.filter(r => {

}


function normTgl(t){

function sortHist(th){

  const col = th.dataset.col;

  if(histSortCol === col){

    histSortDir = histSortDir==='asc' ? 'desc' : 'asc';

  } else {

    histSortCol = col;

    histSortDir = col==='tanggal' ? 'desc' : 'asc';

  }

  // Update header visual

  document.querySelectorAll('table.master-tbl thead th.sortable').forEach(h=>{

    h.classList.remove('sort-asc','sort-desc');

    const icon = h.querySelector('.sort-icon');

    if(icon) icon.textContent = '↕';

  });

  th.classList.add('sort-'+ histSortDir);

  const icon = th.querySelector('.sort-icon');

  if(icon) icon.textContent = histSortDir==='asc' ? '↑' : '↓';


  sortAndRenderHist();

}


function sortAndRenderHist(){

  const col = histSortCol;

  const dir = histSortDir === 'asc' ? 1 : -1;

  const sorted = [...histFiltered].sort((a,b)=>{

    let va = a[col] ?? '';

    let vb = b[col] ?? '';

    // Tanggal sort sebagai string (format yyyy-MM-dd sudah sortable)

    if(typeof va === 'string') va = va.toLowerCase();

    if(typeof vb === 'string') vb = vb.toLowerCase();

    if(va < vb) return -1 * dir;

    if(va > vb) return  1 * dir;

    return 0;

  });

  renderHistRows(sorted);

}
